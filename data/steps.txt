SHOW DATABASES;#顯示所有資料庫
USE <資料庫名稱>;#使用資料庫
SHOW TABLES;#顯示所有的表，需先使用資料庫
CREATE TABLE <表名稱>(欄位);#創建列表
ALTER TABLE <表名稱> ADD <欄名稱 資料類型 ...>;#PRIMARY KEY為主鍵；AUTO_INCREMENT為自動遞增；DEFAUT NOW()預設為現在時間
ALTER TABLE <表名稱> MODIFY <欄名稱>;#修改欄內條件
ALTER TABLE <表名稱> DROP <欄名稱>;#刪除該欄
ALTER TABLE <表名稱> MODIFY <欄名稱 資料類型> FIRST;#將該欄移至第一欄
ALTER TABLE <表名稱> MODIFY <欄名稱 資料類型> AFTER <欄名稱2>;#將<欄名稱>移至<欄名稱2>之後
***SHOW FULL FIELDS FROM <表名稱>#顯示所有欄位
SELECT*FROM <表名稱> ORDER BY <欄位名稱> (DESC);#依照欄位值由大到小排列；DESC則反
SELECT*FROM <表名稱> LIMIT x, y;#列出從第x筆(不含)開始顯示y筆的資料
SELECT*FROM <表名稱> LIMIT x OFFSET y;#從第y筆(不含)資料開始往後列出x筆資料
SELECT*FROM <表名稱> WHERE <欄名稱>='值';#列出與欄名稱的值相同的資料，可加AND增加條件
UPDATE <表名稱> SET <欄名稱> = '需修改的值' WHERE condition;
SELECT COUNT(*) FROM <表名稱>；計算*(所有)的數量，可加Where限定條件
SELECT COUNT(<欄名稱>) FROM <表名稱>；計算<欄名稱>的數量，可加Where限定條件
SELECT SUM(<欄名稱>) FROM <表名稱>；計算<欄名稱>的值加總，可加Where限定條件
SELECT AVG(<欄名稱>) FROM <表名稱>；計算<欄名稱>的值平均，可加Where限定條件
FOREIGN KEY (<欄名稱1>) REFERENCES <表名稱>(<欄名稱2>);如後加欄位設置外鍵需先設置好欄位
SELECT <表名稱>.<欄名稱>, ...#需顯示的欄名稱
FROM <表名稱>
INNER JOIN <表名稱> ON <表名稱>.<欄名稱FK>=<表2名稱>.<欄名稱PK>
